package="ppgmmga"

# compila e installa 
lib="/Library/Frameworks/R.framework/Resources/library"
cd ~/R

# crea package
R --vanilla CMD build --compact-vignettes --force $package

# filename of latest package version
pkgfile=$(ls -t $package*.tar.gz)
pkgfile=$(echo $pkgfile | awk '{print $1}')

# check package
R --vanilla CMD check --as-cran --run-dontrun $pkgfile

# crea pdf manual
# R --vanilla CMD Rd2pdf $package

# remove old package
R --vanilla CMD REMOVE --library="$lib" $package

# install package
R --vanilla CMD INSTALL --library="$lib" $pkgfile
